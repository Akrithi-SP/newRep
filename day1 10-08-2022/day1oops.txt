
import java.util.*;
import java.text.SimpleDateFormat;
import java.text.ParseException;
import java.util.Date;
class Account{
    private int id=0;
    private double balance=0;
    private double annualInterestRate=4.5;
    private Date dateCreated;
    
    Account(){
        
    }
    Account(int i){
        id=i;
        
    }
    
//accessor methods

    public int getId(){
        //System.out.println("Id: "+this.id);
        return id;
    }
    
    public double getBalance(){
        //System.out.println("Balance: "+this.balance);
        return balance;
    }
    
    public double getAnnualInterestRate(){
        //System.out.println("Annual interest rate: "+this.annualInterestRate/100);
        return annualInterestRate/100;
    }
    
    public String getDateCreated(){
        Date date= new Date();
	    SimpleDateFormat formatter = new SimpleDateFormat("dd/MM/yy");
	    String str = formatter.format(date);
        //System.out.println("date: "+str);
        return str;
        
    }
   
//methods    
    public double getMonthlyInterestRate(){
        double annualInterestRate=getAnnualInterestRate();
        double MonthlyInterestRate=annualInterestRate/12;
        return MonthlyInterestRate;
    }
    
    public double getMonthlyInterest(){
        double MonthlyInterestRate=getMonthlyInterestRate();
        double MonthlyInterest=balance*MonthlyInterestRate;
        //System.out.println("monthly interest rate: "+MonthlyInterest);
        return MonthlyInterest;
    }
    
    public void withdraw(double wd){
        if(balance>wd){
            double rem=balance-wd;
            System.out.println("Remaining amount is :"+rem);
        }
        else{
            System.out.println("No sufficient Balance");
        }
    }
    
    public void deposit(double amt){
        double tot=balance+amt;
        System.out.println("Total money is: "+tot);
    }

    
//mutator methods 
    
    public void setId(int newId){
        this.id=newId;
    }
    
    public void setBalance(double newBalance){
        this.balance=newBalance;
    }
    
    public void setAnnualInterestRate(double newAnnualInterestRate){
        this.annualInterestRate=newAnnualInterestRate;
    }
    
}
